# IF
```js 
const x = 10;
const y = 1;
if(x == '10'){
console.log('yeah')
}


if(x === '10'){
console.log('nah')
}

//or
if(x==10 || y==1)

//and
if(x==10 && y==1)
```

# Switch
```js
const color = 'red';

switch(color){
	case 'red':
		console.log('red');
		break;
	case 'blue':
		console.log('blue');
		break;
	default:
		console.log('none');
		break;
}
```
